<script>
  // ✅ PASTE YOUR APPS SCRIPT WEB APP URL HERE
  const API_URL = "https://script.google.com/macros/s/AKfycbyzOAHk-KE3xrBN0dyWb_h_-iqUtIKCZooLPAaqf2t2QAQTMJmp9Chl2gxLLYEzuPIiKQ/exec";

  async function loadData() {
    try {
      const empRes = await fetch(API_URL + "?action=getEmployees");
      const employees = await empRes.json();
      const eSelect = document.getElementById("employee");
      employees.forEach(r => {
        let o = document.createElement("option");
        o.value = r[0]; o.text = r[0];
        eSelect.appendChild(o);
      });

      const shiftRes = await fetch(API_URL + "?action=getShifts");
      const shifts = await shiftRes.json();
      const sSelect = document.getElementById("shift");
      shifts.forEach(r => {
        let o = document.createElement("option");
        o.value = r[0]; o.text = r[0];
        sSelect.appendChild(o);
      });
    } catch (err) {
      document.getElementById("msg").innerHTML = "Failed to load data";
    }
  }

  function send(type) {
    const employee = document.getElementById("employee").value;
    const shift = document.getElementById("shift").value;

    if (!employee || !shift) {
      document.getElementById("msg").innerHTML = "Please select employee and shift";
      return;
    }

    document.getElementById("msg").innerHTML = "Getting location...";

    navigator.geolocation.getCurrentPosition(async function(pos) {
      document.getElementById("msg").innerHTML = "Processing...";
      try {
        const res = await fetch(API_URL, {
          method: "POST",
          body: JSON.stringify({
            action: "submitAttendance",
            employeeName: employee,
            shiftTime: shift,
            actionType: type,
            lat: pos.coords.latitude,
            lng: pos.coords.longitude
          })
        });
        const data = await res.json();
        document.getElementById("msg").innerHTML = data.message;
      } catch (err) {
        document.getElementById("msg").innerHTML = "Submission failed";
      }
    },
    function() {
      document.getElementById("msg").innerHTML = "Location permission denied";
    });
  }

  loadData();
</script>
